void insertionSort(int a[], int n) {

  for (int i = 0; i < n; i++) {

    int j = i;

    // Move a[j] to its correct position in the sorted part of the array

    while (j > 0 && a[j - 1] > a[j]) {

      // Swap if the previous element is greater

      swap(a[j - 1], a[j]);

      j--; // Move left

    }

  }

}
